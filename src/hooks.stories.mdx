# Hooks

List of hooks added by this library.

### Page Title

```ts
import React from 'react';
import { useDocumentTitle } from 'antd-react-extensions';

const Page: React.FC = () => {
  useDocumentTitle('My Example Page');

  return (
    <div>PageContent</div>
  );
};

export default Page;
```

### Component is Mounted

```ts
import React from 'react';
import { useIsMounted } from 'antd-react-extensions';

const Page: React.FC = () => {
  const isMounted = useIsMounted();

  return (
    <div>{isMounted ? 'yes' : 'no'}</div>
  );
};

export default Page;
```

### useState with Callback

```ts
import React from 'react';
import { useStateWithCallback } from 'antd-react-extensions';

const Page: React.FC = () => {
  const value = useStateWithCallback(0, () => {
    console.log('value changed');
  });

  return (
    <div>PageContent</div>
  );
};

export default Page;
```

### Reselect Selector

```ts
import React from 'react';
import { useReselectSelector } from 'antd-react-extensions';
import { createSelector } from 'reselect';

const testSelector = createSelector(
  (state: any) => state.element,
  (_state: any, props: { key: string }) => props.key,
  (state, key) => state[key],
);

const Page: React.FC = () => {
  const value = useReselectSelector(testSelector, { key: a });

  return <div>{JSON.stringify(value)}</div>;
};

export default Page;
```

### Add Class to Body

```ts
import React from 'react';
import { useBodyClass } from 'antd-react-extensions';

const Page: React.FC = () => {
  useBodyClass('class-name');

  return <div>Page</div>;
};

export default Page;
```
